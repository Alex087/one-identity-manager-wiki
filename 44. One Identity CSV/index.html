Public Function CCC_transliteration(ByVal inputValue As String) As String

inputValue = inputValue.ToLower()

Dim SB As New System.Text.StringBuilder

Dim Rus As String() = {"а", "б", "в", "г", "д", "е", "ё", "ж", "з", "и", "й", "к", _

"л", "м", "н", "о", "п", "р", "с", "т", "у", "ф", "х", "ц", "ч", "ш", _

"щ", "ъ", "ы", "ь", "э", "ю", "я"}

Dim Eng As String() = {"a", "b", "v", "g", "d", "e", "e", "zh", "z", "i", "i", _

"k", "l", "m", "n", "o", "p", "r", "s", "t", "u", "f", "kh", "ts", "ch", _

"sh", "shch", "ie", "y", "", "e", "iu", "ia"}

For Each ch In inputValue

If Char.IsLetter(ch) AndAlso AscW(ch) > 192 Then

SB.Append(Eng(Array.IndexOf(Rus, ch.ToString)))

Else

SB.Append(ch)

End If

Next

Return SB.ToString

End Function
