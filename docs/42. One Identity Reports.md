***Условие для учета параметров у Person***

CentralAccount = @account_name or (UID_Department =
dbo.QBM_FCVObjectkeyToElement(N'ColumnValue1',@ObjectKeyBase))

or Exists (select top 1 1 from PersonInDepartment Where
PersonInDepartment.UID_Person = Person.UID_Person and
PersonInDepartment.UID_Department =
dbo.QBM_FCVObjectkeyToElement(N'ColumnValue1',@ObjectKeyBase))

 

***Условие для учета параметров у UNSAccount***

(UID_UNSRoot = dbo.QBM_FCVObjectkeyToElement(N'ColumnValue1',@system))
